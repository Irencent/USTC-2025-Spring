myfun <- function(x, lambda) {
  ifelse(x > lambda, x - lambda,
         ifelse(x < -lambda, x + lambda, 0))
}

  2. 基于上述函数编写求解以下问题的算法（需要用到循环结构） \[ \min_\beta L(\beta) - \frac{1}{2} \|y- X\beta\|^2 + \lambda \sum_{j=1}^3 |\beta_j|, \] 其中y 是一个n维的向量， X是一个 \(n\times 3\) 的矩阵， \(\beta\) 是一个3维的向量。算法描述如下：

    • 在给定当前解 \(\hat\beta\) 时，对每一个维度 \(j\) ,局部最优解为 \[ f\left(\frac{X_j^T(y-X\hat\beta + X_j\hat\beta_j)}{\|X_j\|^2}, \lambda/\|X_j\|^2\right) \]

    • 遍历 \(j=1, 2, 3, 1, 2, 3, \dots\) 直到前后两个解的平方差小于给定的误差（如 1e-6 ）。

最后通过模拟数据验证上述算法是否编写正确并输出最后的估计值，模拟数据代码如下：

coordinate_descent <- function(X, y, lambda, max_iter = 1000, tol = 1e-6) {
  p <- ncol(X)
  beta <- rep(0, p)  # 初始化 beta 为全零
  iter <- 0
  converged <- FALSE
  
  while (iter < max_iter && !converged) {
    beta_old <- beta
    
    # 遍历每个维度更新系数
    for (j in 1:p) {
      X_j <- X[, j]
      # 计算残差（移除当前维度的贡献）
      r <- y - X %*% beta + X_j * beta[j]
      # 计算局部最优解
      a <- sum(X_j * r) / sum(X_j^2)
      lambda_scaled <- lambda / sum(X_j^2)
      beta[j] <- myfun(a, lambda_scaled)
    }
    
    # 检查收敛条件
    if (sqrt(sum((beta - beta_old)^2)) < tol) converged <- TRUE
    iter <- iter + 1
  }
  beta
}
